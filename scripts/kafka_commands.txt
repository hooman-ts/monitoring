		./bin/kafka-configs.sh --alter --add-config retention.ms=10000 --bootstrap-server=0.0.0.0:9092 --topic purge-scenario

# get a specific offset
	bin/kafka-console-consumer.sh --bootstrap-server localhost:9092 --topic mytopic --offset 10 --partition 0

# List of consumer group:
	./bin/kafka-consumer-groups.sh --bootstrap-server kafka-node01:9092 --all-groups -describe
# or
	./bin/kafka-consumer-groups.sh --bootstrap-server kafka-node01:9092 --group myConsumerGroup --describe

# make sure that the consumer group is not active
# Reset offset to the latest or the earliest:
	./bin/kafka-consumer-groups.sh --bootstrap-server kafka-node01:9092 --group myConsumerGroup --reset-offsets --to-earliest --topic my_topic -execute
	./bin/kafka-consumer-groups.sh --bootstrap-server kafka-node01:9092 --group myConsumerGroup --reset-offsets --to-latest --topic my_topic -execute

# Reset the consumer offset to specify a time:
	./bin/kafka-consumer-groups.sh --bootstrap-server localhost:9092 --group myConsumerGroup --reset-offsets --to-datetime 2020-12-20T00:00:00.000 --topic my_topic --execute
